> ## CHANGE 5 — Pricing Calculation for Moving Flow
>
> Add this calculation to `lib/pricing.ts`:
>
> ```ts
> export function getMovingQuote(
>   sizeIdx: number,
>   tier: Tier,
>   durationDays: number | null,
>   route: 'clutter' | 'flex'
> ): {
>   monthlyRate: number
>   estimatedTotal: number | null
>   pickupCost: number
>   deliveryCost: number
>   tierAdjustment: number
>   planLabel: string
> } {
>   const p = pricing[sizeIdx]
>   const mult = tierMultipliers[tier]
>
>   if (route === 'flex') {
>     // Flat move quote — no monthly storage rate
>     const baseFlexQuote = [299, 399, 499, 649, 799, 949, 1099, 1299][sizeIdx]
>     const tierAdj = tier === 'prepacked' ? -50 : tier === 'youload' ? -100 : 0
>     return {
>       monthlyRate: 0,
>       estimatedTotal: baseFlexQuote + tierAdj,
>       pickupCost: 0,
>       deliveryCost: 0,
>       tierAdjustment: tierAdj,
>       planLabel: 'Flex Move'
>     }
>   }
>
>   // Clutter route
>   const months = durationDays ? Math.ceil(durationDays / 30) : null
>   const plan = !months || months >= 8 ? 'longhaul'
>              : months >= 4 ? 'committed'
>              : 'flexible'
>   const monthlyRate = Math.round(
>     (plan === 'longhaul' ? p.eight : plan === 'committed' ? p.four : p.m2m) * mult
>   )
>   const pickupCost = plan === 'flexible' ? 149 : 0
>   const deliveryCost = plan === 'flexible' ? 149 : 0
>   const estimatedTotal = months
>     ? monthlyRate * months + pickupCost + deliveryCost
>     : null
>
>   return {
>     monthlyRate,
>     estimatedTotal,
>     pickupCost,
>     deliveryCost,
>     tierAdjustment: 0,
>     planLabel: plan === 'longhaul' ? 'Long Haul'
>              : plan === 'committed' ? 'Committed'
>              : 'Flexible'
>   }
> }
> ```
>
> The Outcome screen calls `getMovingQuote(state.sizeIdx, state.tier, durationDays, state.movingRoute)` and renders all values from the returned object dynamically.
